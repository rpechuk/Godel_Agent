    response = agent.action_call_json_format_llm(
        model="gpt-3.5-turbo", 
        messages=messages, 
        temperature=0.5, 
        num_of_response=1,
        role="read comprehension expert",

        requirements=(
            "1. Please explain step by step.\n"

            "3. The answer MUST be a concise string.\n"
        ).strip(), 
    )

    return_dict["answer"] = str(return_dict.get("answer", ""))
    return return_dict
